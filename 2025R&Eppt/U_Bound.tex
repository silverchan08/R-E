\section{Upper Bounds of Arc Index}

\begin{frame}{Upper Bounds of Arc Index}
	\begin{thm}[Lee et al., 2018]
		Let $G$ be any spatial graph. Then,
        \[ \alpha(G) \le c(G) + e + b \]
		where $e$ and $b$ are the number of edges and bouquet cut, respectively.
	\end{thm}

	\begin{corollary}
		Let $T$ be any non-trivial prime $\theta$-curve or handcuff graph. Then,
        \[ \alpha(T) \le c(T) + 3. \]
	\end{corollary}
\end{frame}
