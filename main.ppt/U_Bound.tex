\section{Lower and Upper Bounds of Arc Index}

\begin{frame}{Upper Bounds of Arc Index}
	\begin{thm}
		Let $T$ be any $\theta$-curve. Then,
        \[ \alpha(T) \le c(T) + 3. \]
	\end{thm}

    \begin{thm}
        Let $H$ be any handcuff graph. Then,
        \[ \alpha(H) \le c(H) + 5. \]
        Especially, if the constituent link of $H$ is non-split,
        \[ \alpha(H) \le c(H) + 3. \]
    \end{thm}
\end{frame}

\begin{frame}{Upper Bounds of Arc Index}
	\begin{thm}
		Let $T$ be any non-trivial prime $\theta$-curve or handcuff graph. Then,
        \[ \alpha(T) \le c(T) + 3. \]
	\end{thm}
\end{frame}
