\begin{frame}{THC-Cromwell matrix}

		\item The \term{Cromwell matrix} of a knot is an $n\times n$ binary matrix each of whose rows and columns has exactly two 1s.

\begin{frame}{H-deletion of THC-Cromwell matrices}
	\begin{itemize}
		\item The \term{vertex-row} of THC-Cromwell matrix $M$ is a row which contains three 1s, $M_{ia},M_{ib},M_{ic}$, where $a<b<c$, as its elements.
		\item The \term{side-column} of THC-Cromwell matrix $M$ is a column which contains the leftmost 1 of vertex-row $(M_{ia})$ or the rightmost 1 of vertex row $(M_{ic})$.
		\item The \term{H-deletion} Matrix of the THC-Cromwell
 		matrix G is $(n-1)\times(n-1)$ matrix\\ which deleted vertex-row and its two side-columns from the 
 		matrix G.
	\end{itemize}

\begin{frame}{Determinants of the THC-Cromwell matrices}
	\begin{newthm}
		Let $M_G$ be a THC-Cromwell matrix of $\theta$-curve or handcuff graph $G$. \\
		\begin{itemize}
			\item det*($M_G$) = $\pm1$ $\iff$ $M_G$ represents $\theta$-curve
			\item det*($M_G$) = $0$ or $\pm2$ $\iff$ $M_G$ represents handcuff graph
		\end{itemize}
		 where det*($M_G$) = determinant of H-deletion matrix of $M_G$
	\end{newthm}
	\mypf
